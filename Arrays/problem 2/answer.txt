public class Solution {
    public int[] GetConcatenation(int[] nums) {
        int[] ans = new int[2*nums.Length];
        for(int i = 0; i < 2*nums.Length; i++){
          if(i < nums.Length)
            ans[i] =  nums[i];
          else
            ans[i] =  nums[i-nums.Length];
        }
        return ans;
    }
}
